**Goal:** Add all high and medium impact openpyxl-compatible features (conditional formatting, data validation, auto-filter, hyperlinks, comments, page setup, named ranges, images, sheet protection) plus a batch cell API for speed improvement.

### Task 1: Batch Cell API (Speed Improvement)

**Files:**

- Modify: `python/openpyxl_rust/worksheet.py` — add `append()` method

**Rust changes (lib.rs):** Add to `#[pymethods] impl RustWorkbook`:

```rust
fn set_rows_batch(&mut self, sheet: usize, start_row: u32, rows: &Bound<'_, PyList>) -> PyResult<()> {
    let sd = self.sheets.get_mut(sheet)
        .ok_or_else(|| pyo3::exceptions::PyIndexError::new_err("Sheet index out of range"))?;
    for (r_offset, row_obj) in rows.iter().enumerate() {
        let row_list: &Bound<'_, PyList> = row_obj.downcast()?;
        let row = start_row + r_offset as u32;
        for (c_idx, val) in row_list.iter().enumerate() {
            let col = c_idx as u16;
            let key = (row, col);
            let cell_data = if val.is_none() {
                CellData::Empty
            } else if let Ok(b) = val.extract::<bool>() {
                CellData::Boolean(b)
            } else if let Ok(n) = val.extract::<f64>() {
                CellData::Number(n)
            } else if let Ok(s) = val.extract::<String>() {
                if s.starts_with('=') { CellData::Formula(s) } else { CellData::String(s) }
            } else {
                CellData::Empty
            };
            if let Some(cv) = sd.cells.get_mut(&key) {
                cv.value = cell_data;
            } else {
                sd.cells.insert(key, CellValue { value: cell_data, format_json: None });
            }
        }
    }
    Ok(())
}
```

**Python changes (worksheet.py):** Add method:

```python
def append(self, iterable):
    """Append a row of values. Matches openpyxl's ws.append() API."""
    # Find next empty row
    if self._cells:
        row = max(r for r, _ in self._cells.keys()) + 1
    else:
        row = 1
    for col_idx, value in enumerate(iterable, 1):
        self.cell(row=row, column=col_idx, value=value)
```

Also add `_current_row` tracker for efficient append and a batch-aware path:

```python
def append_rows(self, rows_data):
    """Append multiple rows at once via Rust batch API for speed."""
    if self._workbook is None or self._sheet_idx is None:
        for row in rows_data:
            self.append(row)
        return
    if self._cells:
        start_row = max(r for r, _ in self._cells.keys()) + 1
    else:
        start_row = 1
    # Push to Rust in batch
    self._workbook._rust_wb.set_rows_batch(self._sheet_idx, start_row - 1, rows_data)
    # Also store in Python cells dict for format access
    for r_offset, row in enumerate(rows_data):
        for c_idx, value in enumerate(row):
            r = start_row + r_offset
            c = c_idx + 1
            self._cells[(r, c)] = Cell(row=r, column=c, value=value)
```

```python
def bench_large_data_rust_batch(path, rows=100_000, cols=10):
    """100k rows x 10 cols using batch API."""
    from openpyxl_rust import Workbook
    wb = Workbook()
    ws = wb.active
    ws.title = "Data"
    batch = []
    for r in range(1, rows + 1):
        row = []
        for c in range(1, cols + 1):
            if c % 3 == 0:
                row.append(f"str_{r}_{c}")
            elif c % 3 == 1:
                row.append(r * c * 1.1)
            else:
                row.append(r % 2 == 0)
        batch.append(row)
    ws.append_rows(batch)
    wb.save(path)
```
