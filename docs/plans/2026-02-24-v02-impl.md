```python
class Alignment:
    def __init__(self, horizontal=None, vertical=None, wrap_text=False,
                 shrink_to_fit=False, indent=0, text_rotation=0):
        self.horizontal = horizontal
        self.vertical = vertical
        self.wrap_text = wrap_text
        self.shrink_to_fit = shrink_to_fit
        self.indent = indent
        self.text_rotation = text_rotation

    def __eq__(self, other):
        if not isinstance(other, Alignment):
            return NotImplemented
        return (self.horizontal == other.horizontal
                and self.vertical == other.vertical
                and self.wrap_text == other.wrap_text
                and self.shrink_to_fit == other.shrink_to_fit
                and self.indent == other.indent
                and self.text_rotation == other.text_rotation)

    def __repr__(self):
        return (f"Alignment(horizontal={self.horizontal!r}, vertical={self.vertical!r}, "
                f"wrap_text={self.wrap_text})")
```

```python
class Side:

        self.style = style
        self.color = color

    def __eq__(self, other):
        if not isinstance(other, Side):
            return NotImplemented
        return self.style == other.style and self.color == other.color

    def __repr__(self):
        return f"Side(style={self.style!r}, color={self.color!r})"


class Border:
    def __init__(self, left=None, right=None, top=None, bottom=None):
        self.left = left or Side()
        self.right = right or Side()
        self.top = top or Side()
        self.bottom = bottom or Side()

    def __eq__(self, other):
        if not isinstance(other, Border):
            return NotImplemented
        return (self.left == other.left and self.right == other.right
                and self.top == other.top and self.bottom == other.bottom)

    def __repr__(self):
        return f"Border(left={self.left!r}, right={self.right!r}, top={self.top!r}, bottom={self.bottom!r})"
```

Replace the `make_format` function with this expanded version that handles the full format dict (font + alignment + border + fill). Also rename it to `build_format` since it now handles more than fonts:

```rust
fn build_format(fmt_dict: &Bound<'_, PyDict>) -> Result<Format, PyErr> {
    let mut fmt = Format::new();

    // Font
    if let Ok(Some(font_obj)) = fmt_dict.get_item("font") {
        if !font_obj.is_none() {
            let font: &Bound<'_, PyDict> = font_obj.downcast()?;
            if let Ok(Some(bold)) = font.get_item("bold") {
                if bold.extract::<bool>()? {
                    fmt = fmt.set_bold();
                }
            }
            if let Ok(Some(italic)) = font.get_item("italic") {
                if italic.extract::<bool>()? {
                    fmt = fmt.set_italic();
                }
            }
            if let Ok(Some(underline)) = font.get_item("underline") {
                let ul: String = underline.extract()?;
                if ul == "single" {
                    fmt = fmt.set_underline(rust_xlsxwriter::FormatUnderline::Single);
                } else if ul == "double" {
                    fmt = fmt.set_underline(rust_xlsxwriter::FormatUnderline::Double);
                }
            }
            if let Ok(Some(name)) = font.get_item("name") {
                let n: String = name.extract()?;
                fmt = fmt.set_font_name(&n);
            }
            if let Ok(Some(size)) = font.get_item("size") {
                let s: f64 = size.extract()?;
                fmt = fmt.set_font_size(s);
            }
            if let Ok(Some(color)) = font.get_item("color") {
                let c: String = color.extract()?;
                if let Ok(rgb) = u32::from_str_radix(&c, 16) {
                    fmt = fmt.set_font_color(rust_xlsxwriter::Color::from(rgb));
                }
            }
        }
    }

    // Alignment
    if let Ok(Some(align_obj)) = fmt_dict.get_item("alignment") {
        let align: &Bound<'_, PyDict> = align_obj.downcast()?;
        if let Ok(Some(h)) = align.get_item("horizontal") {
            let hs: String = h.extract()?;
            let a = match hs.as_str() {
                "center" => rust_xlsxwriter::FormatAlign::Center,
                "right" => rust_xlsxwriter::FormatAlign::Right,
                "left" => rust_xlsxwriter::FormatAlign::Left,
                "fill" => rust_xlsxwriter::FormatAlign::Fill,
                "justify" => rust_xlsxwriter::FormatAlign::Justify,
                "centerContinuous" | "center_continuous" => rust_xlsxwriter::FormatAlign::CenterAcross,
                "distributed" => rust_xlsxwriter::FormatAlign::Distributed,
                "general" => rust_xlsxwriter::FormatAlign::General,
                _ => rust_xlsxwriter::FormatAlign::General,
            };
            fmt = fmt.set_align(a);
        }
        if let Ok(Some(v)) = align.get_item("vertical") {
            let vs: String = v.extract()?;
            let a = match vs.as_str() {
                "center" => rust_xlsxwriter::FormatAlign::VerticalCenter,
                "top" => rust_xlsxwriter::FormatAlign::Top,
                "bottom" => rust_xlsxwriter::FormatAlign::Bottom,
                "justify" => rust_xlsxwriter::FormatAlign::VerticalJustify,
                "distributed" => rust_xlsxwriter::FormatAlign::VerticalDistributed,
                _ => rust_xlsxwriter::FormatAlign::Bottom,
            };
            fmt = fmt.set_align(a);
        }
        if let Ok(Some(wt)) = align.get_item("wrap_text") {
            if wt.extract::<bool>()? {
                fmt = fmt.set_text_wrap();
            }
        }
        if let Ok(Some(sf)) = align.get_item("shrink_to_fit") {
            if sf.extract::<bool>()? {
                fmt = fmt.set_shrink();
            }
        }
        if let Ok(Some(ind)) = align.get_item("indent") {
            let i: u8 = ind.extract()?;
            if i > 0 {
                fmt = fmt.set_indent(i);
            }
        }
        if let Ok(Some(rot)) = align.get_item("text_rotation") {
            let r: i16 = rot.extract()?;
            if r != 0 {
                fmt = fmt.set_rotation(r);
            }
        }
    }

    // Border
    if let Ok(Some(border_obj)) = fmt_dict.get_item("border") {
        let border: &Bound<'_, PyDict> = border_obj.downcast()?;

        fn parse_border_style(s: &str) -> rust_xlsxwriter::FormatBorder {
            match s {
                "thin" => rust_xlsxwriter::FormatBorder::Thin,
                "medium" => rust_xlsxwriter::FormatBorder::Medium,
                "thick" => rust_xlsxwriter::FormatBorder::Thick,
                "dashed" => rust_xlsxwriter::FormatBorder::Dashed,
                "dotted" => rust_xlsxwriter::FormatBorder::Dotted,
                "double" => rust_xlsxwriter::FormatBorder::Double,
                "hair" => rust_xlsxwriter::FormatBorder::Hair,
                "mediumDashed" => rust_xlsxwriter::FormatBorder::MediumDashed,
                "dashDot" => rust_xlsxwriter::FormatBorder::DashDot,
                "mediumDashDot" => rust_xlsxwriter::FormatBorder::MediumDashDot,
                "dashDotDot" => rust_xlsxwriter::FormatBorder::DashDotDot,
                "mediumDashDotDot" => rust_xlsxwriter::FormatBorder::MediumDashDotDot,
                "slantDashDot" => rust_xlsxwriter::FormatBorder::SlantDashDot,
                _ => rust_xlsxwriter::FormatBorder::Thin,
            }
        }

        fn parse_color(c: &str) -> Option<rust_xlsxwriter::Color> {
            u32::from_str_radix(c, 16).ok().map(rust_xlsxwriter::Color::from)
        }

        if let Ok(Some(left_obj)) = border.get_item("left") {
            let left: &Bound<'_, PyDict> = left_obj.downcast()?;
            if let Ok(Some(style)) = left.get_item("style") {
                let s: String = style.extract()?;
                fmt = fmt.set_border_left(parse_border_style(&s));
            }
            if let Ok(Some(color)) = left.get_item("color") {
                let c: String = color.extract()?;
                if let Some(clr) = parse_color(&c) {
                    fmt = fmt.set_border_left_color(clr);
                }
            }
        }
        if let Ok(Some(right_obj)) = border.get_item("right") {
            let right: &Bound<'_, PyDict> = right_obj.downcast()?;
            if let Ok(Some(style)) = right.get_item("style") {
                let s: String = style.extract()?;
                fmt = fmt.set_border_right(parse_border_style(&s));
            }
            if let Ok(Some(color)) = right.get_item("color") {
                let c: String = color.extract()?;
                if let Some(clr) = parse_color(&c) {
                    fmt = fmt.set_border_right_color(clr);
                }
            }
        }
        if let Ok(Some(top_obj)) = border.get_item("top") {
            let top: &Bound<'_, PyDict> = top_obj.downcast()?;
            if let Ok(Some(style)) = top.get_item("style") {
                let s: String = style.extract()?;
                fmt = fmt.set_border_top(parse_border_style(&s));
            }
            if let Ok(Some(color)) = top.get_item("color") {
                let c: String = color.extract()?;
                if let Some(clr) = parse_color(&c) {
                    fmt = fmt.set_border_top_color(clr);
                }
            }
        }
        if let Ok(Some(bottom_obj)) = border.get_item("bottom") {
            let bottom: &Bound<'_, PyDict> = bottom_obj.downcast()?;
            if let Ok(Some(style)) = bottom.get_item("style") {
                let s: String = style.extract()?;
                fmt = fmt.set_border_bottom(parse_border_style(&s));
            }
            if let Ok(Some(color)) = bottom.get_item("color") {
                let c: String = color.extract()?;
                if let Some(clr) = parse_color(&c) {
                    fmt = fmt.set_border_bottom_color(clr);
                }
            }
        }
    }

    // Fill
    if let Ok(Some(fill_obj)) = fmt_dict.get_item("fill") {
        let fill: &Bound<'_, PyDict> = fill_obj.downcast()?;
        if let Ok(Some(ft)) = fill.get_item("fill_type") {
            let fts: String = ft.extract()?;
            let pattern = match fts.as_str() {
                "solid" => rust_xlsxwriter::FormatPattern::Solid,
                "darkGray" | "darkDown" => rust_xlsxwriter::FormatPattern::DarkGray,
                "mediumGray" => rust_xlsxwriter::FormatPattern::MediumGray,
                "lightGray" => rust_xlsxwriter::FormatPattern::LightGray,
                "gray125" => rust_xlsxwriter::FormatPattern::Gray125,
                "gray0625" => rust_xlsxwriter::FormatPattern::Gray0625,
                _ => rust_xlsxwriter::FormatPattern::Solid,
            };
            fmt = fmt.set_pattern(pattern);
        }
        if let Ok(Some(sc)) = fill.get_item("start_color") {
            let c: String = sc.extract()?;
            if let Ok(rgb) = u32::from_str_radix(&c, 16) {
                fmt = fmt.set_background_color(rust_xlsxwriter::Color::from(rgb));
            }
        }
        if let Ok(Some(ec)) = fill.get_item("end_color") {
            let c: String = ec.extract()?;
            if let Ok(rgb) = u32::from_str_radix(&c, 16) {
                fmt = fmt.set_foreground_color(rust_xlsxwriter::Color::from(rgb));
            }
        }
    }

    // Number format
    if let Ok(Some(nf)) = fmt_dict.get_item("number_format") {
        let nf_str: String = nf.extract()?;
        if nf_str != "General" {
            fmt = fmt.set_num_format(&nf_str);
        }
    }

    Ok(fmt)
}
```

```rust
fn date_to_excel_serial(year: i32, month: i32, day: i32) -> f64 {
    // Convert date to Excel serial number (days since 1899-12-30)
    // Includes the Lotus 1-2-3 bug: Excel thinks 1900 was a leap year
    let mut y = year;
    let mut m = month;
    if m <= 2 {
        y -= 1;
        m += 12;
    }
    let a = y / 100;
    let b = 2 - a + a / 4;
    let jd = (365.25 * (y + 4716) as f64) as i64
        + (30.6001 * (m + 1) as f64) as i64
        + day as i64
        + b as i64
        - 1524;
    // Excel epoch: Jan 1 1900 = serial 1, but Julian day of Dec 30 1899
    let excel_epoch_jd = 2415019; // Julian day of 1899-12-30
    let mut serial = (jd - excel_epoch_jd) as f64;
    // Lotus 1-2-3 bug: if date is after Feb 28 1900, add 1
    if serial > 59.0 {
        serial += 1.0;
    }
    serial
}
```

```rust
use calamine::{open_workbook, Reader, Xlsx, Data};

#[pyfunction]
fn _load_workbook(py: Python<'_>, path: &str) -> PyResult<PyObject> {
    let mut workbook: Xlsx<_> = open_workbook(path)
        .map_err(|e| pyo3::exceptions::PyIOError::new_err(e.to_string()))?;

    let sheet_names: Vec<String> = workbook.sheet_names().to_vec();

    let result = PyDict::new(py);
    let names_list = PyList::new(py, &sheet_names)?;
    result.set_item("sheet_names", names_list)?;

    let sheets_dict = PyDict::new(py);

    for name in &sheet_names {
        let range = workbook.worksheet_range(name)
            .map_err(|e| pyo3::exceptions::PyRuntimeError::new_err(e.to_string()))?;

        let (num_rows, num_cols) = range.get_size();
        let rows_list = PyList::empty(py);

        for r in 0..num_rows {
            let row_list = PyList::empty(py);
            for c in 0..num_cols {
                let cell = range.get((r, c));
                let py_val: PyObject = match cell {
                    Some(Data::String(s)) => s.into_pyobject(py)?.into_any().unbind(),
                    Some(Data::Float(f)) => {
                        // Return as int if it's a whole number
                        if *f == (*f as i64) as f64 && f.is_finite() {
                            (*f as i64).into_pyobject(py)?.into_any().unbind()
                        } else {
                            f.into_pyobject(py)?.into_any().unbind()
                        }
                    }
                    Some(Data::Int(i)) => i.into_pyobject(py)?.into_any().unbind(),
                    Some(Data::Bool(b)) => b.into_pyobject(py)?.into_any().unbind(),
                    Some(Data::DateTime(dt)) => {
                        // Convert ExcelDateTime to ISO string, let Python parse it
                        let s = dt.to_string();
                        s.into_pyobject(py)?.into_any().unbind()
                    }
                    Some(Data::DateTimeIso(s)) => s.into_pyobject(py)?.into_any().unbind(),
                    Some(Data::DurationIso(s)) => s.into_pyobject(py)?.into_any().unbind(),
                    Some(Data::Error(e)) => format!("#ERROR: {:?}", e).into_pyobject(py)?.into_any().unbind(),
                    Some(Data::Empty) | None => py.None(),
                };
                row_list.append(py_val)?;
            }
            rows_list.append(row_list)?;
        }

        sheets_dict.set_item(name.as_str(), rows_list)?;
    }

    result.set_item("sheets", sheets_dict)?;
    Ok(result.into())
}
```

```rust
                    CellData::String(s) => s.into_pyobject(py).unwrap().into_any().unbind(),
                    CellData::Number(n) => n.into_pyobject(py).unwrap().into_any().unbind(),
                    CellData::Boolean(b) => b.into_pyobject(py).unwrap().into_any().unbind(),
                    CellData::Formula(f) => f.into_pyobject(py).unwrap().into_any().unbind(),
                    CellData::DateTime { serial, .. } => serial.into_pyobject(py).unwrap().into_any().unbind(),
                    CellData::Empty => py.None(),
                };
            }
        }
        py.None()
    }

    fn set_column_width(&mut self, sheet: usize, col: u16, width: f64) {
        if let Some(s) = self.sheets.get_mut(sheet) {
            s.column_widths.insert(col, width);

    fn set_row_height(&mut self, sheet: usize, row: u32, height: f64) {

    fn save(&self, py: Python<'_>, path: Option<&str>) -> PyResult<PyObject> {
        let mut workbook = Workbook::new();

        for sheet_data in &self.sheets {
            let worksheet = workbook.add_worksheet();
            worksheet.set_name(&sheet_data.title).map_err(xlsx_err)?;

            for ((row, col), cell) in &sheet_data.cells {
                // Build format if present
                let (fmt, has_format) = if let Some(fmt_map) = &cell.format_dict {
                    if let Some(FormatValue::Str(json)) = fmt_map.get("__raw_json__") {
                        // Parse JSON and build format using existing build_format logic
                        // For now, use a Python-side approach: parse the JSON string
                        // We'll use a simpler approach: call build_format_from_json
                        (build_format_from_json(json)?, true)
                    } else {
                        (Format::new(), false)
                    }
                } else {
                    (Format::new(), false)
                };

                match &cell.value {
                    CellData::String(s) => {
                        if has_format {
                            worksheet.write_string_with_format(*row, *col, s, &fmt).map_err(xlsx_err)?;
                        } else {
                            worksheet.write_string(*row, *col, s).map_err(xlsx_err)?;
                        }
                    }
                    CellData::Number(n) => {
                        if has_format {
                            worksheet.write_number_with_format(*row, *col, *n, &fmt).map_err(xlsx_err)?;
                        } else {
                            worksheet.write_number(*row, *col, *n).map_err(xlsx_err)?;
                        }
                    }
                    CellData::Boolean(b) => {
                        if has_format {
                            worksheet.write_boolean_with_format(*row, *col, *b, &fmt).map_err(xlsx_err)?;
                        } else {
                            worksheet.write_boolean(*row, *col, *b).map_err(xlsx_err)?;
                        }
                    }
                    CellData::Formula(f) => {
                        let formula = Formula::new(f);
                        if has_format {
                            worksheet.write_formula_with_format(*row, *col, formula, &fmt).map_err(xlsx_err)?;
                        } else {
                            worksheet.write_formula(*row, *col, formula).map_err(xlsx_err)?;
                        }
                    }
                    CellData::DateTime { serial, .. } => {
                        if has_format {
                            worksheet.write_number_with_format(*row, *col, *serial, &fmt).map_err(xlsx_err)?;
                        } else {
                            worksheet.write_number(*row, *col, *serial).map_err(xlsx_err)?;
                        }
                    }
                    CellData::Empty => {
                        if has_format {
                            worksheet.write_blank(*row, *col, &fmt).map_err(xlsx_err)?;
                        }
                    }
                }
            }

            for (col, width) in &sheet_data.column_widths {
                worksheet.set_column_width(*col, *width).map_err(xlsx_err)?;
            }
            for (row, height) in &sheet_data.row_heights {
                worksheet.set_row_height(*row, *height).map_err(xlsx_err)?;
            }
            if let Some((row, col)) = sheet_data.freeze_panes {
                worksheet.set_freeze_panes(row, col).map_err(xlsx_err)?;
            }
            for (r1, c1, r2, c2) in &sheet_data.merged_ranges {
                worksheet.merge_range(*r1, *c1, *r2, *c2, "", &Format::new()).map_err(xlsx_err)?;
            }
        }

        if let Some(p) = path {
            workbook.save(p).map_err(xlsx_err)?;
            Ok(py.None())
        } else {
            let buf = workbook.save_to_buffer().map_err(xlsx_err)?;
            Ok(PyBytes::new(py, &buf).into())
        }
    }
}
```

```python
class Workbook:
    def __init__(self):
        from openpyxl_rust._openpyxl_rust import RustWorkbook
        self._rust_wb = RustWorkbook()
        self._sheets = [Worksheet(title="Sheet", workbook=self, sheet_idx=0)]

    @property
    def active(self):
        return self._sheets[0] if self._sheets else None

    def create_sheet(self, title=None):
        title = title or f"Sheet{len(self._sheets) + 1}"
        idx = self._rust_wb.add_sheet(title)
        ws = Worksheet(title=title, workbook=self, sheet_idx=idx)
        self._sheets.append(ws)
        return ws

    @property
    def sheetnames(self):
        return [s.title for s in self._sheets]

    def __getitem__(self, name):
        for s in self._sheets:
            if s.title == name:
                return s
        raise KeyError(f"Worksheet '{name}' not found")

    def save(self, filename):
        # Flush any pending format data to Rust
        for ws in self._sheets:
            ws._flush_formats()

        if isinstance(filename, (str, bytes, os.PathLike)):
            self._rust_wb.save(str(filename))
        else:
            result_bytes = self._rust_wb.save(None)
            filename.write(result_bytes)
```

```python
# python/openpyxl_rust/worksheet.py
import re
from datetime import datetime, date
from openpyxl_rust.cell import Cell, _col_letter


def _parse_cell_ref(ref_str):
    """Parse 'A1' -> (row, col) as 1-based integers."""
    m = re.match(r"^([A-Z]+)(\d+)$", ref_str.upper())
    if not m:
        raise ValueError(f"Invalid cell reference: {ref_str}")
    letters, row_str = m.group(1), m.group(2)
    col = 0
    for ch in letters:
        col = col * 26 + (ord(ch) - 64)
    return int(row_str), col


def _date_to_excel_serial(year, month, day):
    """Convert date to Excel serial number (days since 1899-12-30)."""
    import math
    y, m = year, month
    if m <= 2:
        y -= 1
        m += 12
    a = y // 100
    b = 2 - a + a // 4
    jd = int(365.25 * (y + 4716)) + int(30.6001 * (m + 1)) + day + b - 1524
    excel_epoch_jd = 2415019  # Julian day of 1899-12-30
    serial = jd - excel_epoch_jd
    if serial > 59:
        serial += 1  # Lotus 1-2-3 bug
    return float(serial)


class ColumnDimension:
    def __init__(self):
        self.width = None


class RowDimension:
    def __init__(self):
        self.height = None


class _ColumnDimensionsDict:
    def __init__(self, worksheet=None):
        self._dims = {}
        self._worksheet = worksheet

    def __getitem__(self, key):
        if key not in self._dims:
            self._dims[key] = ColumnDimension()
        return self._dims[key]

    def items(self):
        return self._dims.items()


class _RowDimensionsDict:
    def __init__(self, worksheet=None):
        self._dims = {}
        self._worksheet = worksheet

    def __getitem__(self, key):
        if key not in self._dims:
            self._dims[key] = RowDimension()
        return self._dims[key]

    def items(self):
        return self._dims.items()


class Worksheet:
    def __init__(self, title="Sheet", workbook=None, sheet_idx=0):
        self._title = title
        self._workbook = workbook
        self._sheet_idx = sheet_idx
        self._cells = {}  # local cache: (row, col) -> Cell
        self.column_dimensions = _ColumnDimensionsDict(self)
        self.row_dimensions = _RowDimensionsDict(self)
        self.freeze_panes = None
        self.merged_cell_ranges = []

    @property
    def title(self):
        return self._title

    @title.setter
    def title(self, value):
        self._title = value
        if self._workbook and self._workbook._rust_wb:
            self._workbook._rust_wb.set_sheet_title(self._sheet_idx, value)

    def _set_rust_value(self, row, col, value):
        """Push a value to the Rust workbook (0-based row/col)."""
        if not self._workbook or not self._workbook._rust_wb:
            return
        rb = self._workbook._rust_wb
        r, c = row - 1, col - 1
        if value is None:
            rb.set_cell_empty(self._sheet_idx, r, c)
        elif isinstance(value, bool):
            rb.set_cell_boolean(self._sheet_idx, r, c, value)
        elif isinstance(value, datetime):
            serial = _date_to_excel_serial(value.year, value.month, value.day)
            time_frac = (value.hour * 3600 + value.minute * 60 + value.second) / 86400.0
            rb.set_cell_datetime(self._sheet_idx, r, c, serial + time_frac, False)
        elif isinstance(value, date):
            serial = _date_to_excel_serial(value.year, value.month, value.day)
            rb.set_cell_datetime(self._sheet_idx, r, c, serial, True)
        elif isinstance(value, (int, float)):
            rb.set_cell_number(self._sheet_idx, r, c, float(value))
        elif isinstance(value, str):
            rb.set_cell_string(self._sheet_idx, r, c, value)

    def __setitem__(self, key, value):
        row, col = _parse_cell_ref(key)
        if (row, col) in self._cells:
            self._cells[(row, col)].value = value
        else:
            self._cells[(row, col)] = Cell(row=row, column=col, value=value)
        self._set_rust_value(row, col, value)

    def __getitem__(self, key):
        row, col = _parse_cell_ref(key)
        if (row, col) not in self._cells:
            self._cells[(row, col)] = Cell(row=row, column=col)
        return self._cells[(row, col)]

    def cell(self, row, column, value=None):
        if (row, column) in self._cells:
            cell = self._cells[(row, column)]
            if value is not None:
                cell.value = value
                self._set_rust_value(row, column, value)
            return cell
        c = Cell(row=row, column=column, value=value)
        self._cells[(row, column)] = c
        if value is not None:
            self._set_rust_value(row, column, value)
        return c

    def merge_cells(self, range_string):
        parts = range_string.split(":")
        if len(parts) != 2:
            raise ValueError(f"Invalid merge range: {range_string}")
        self.merged_cell_ranges.append((parts[0].upper(), parts[1].upper()))
        if self._workbook and self._workbook._rust_wb:
            r1, c1 = _parse_cell_ref(parts[0])
            r2, c2 = _parse_cell_ref(parts[1])
            self._workbook._rust_wb.add_merge_range(
                self._sheet_idx, r1 - 1, c1 - 1, r2 - 1, c2 - 1)

    def _flush_formats(self):
        """Flush all cell formats and structural data to Rust before save."""
        import json
        if not self._workbook or not self._workbook._rust_wb:
            return
        rb = self._workbook._rust_wb

        for (row, col), cell in self._cells.items():
            fmt = {}
            if cell.font is not None:
                font_data = {"bold": cell.font.bold, "italic": cell.font.italic,
                             "name": cell.font.name, "size": cell.font.size}
                if cell.font.underline is not None:
                    font_data["underline"] = cell.font.underline
                if cell.font.color is not None:
                    font_data["color"] = cell.font.color
                fmt["font"] = font_data
            if cell.alignment is not None:
                align_data = {}
                if cell.alignment.horizontal is not None:
                    align_data["horizontal"] = cell.alignment.horizontal
                if cell.alignment.vertical is not None:
                    align_data["vertical"] = cell.alignment.vertical
                if cell.alignment.wrap_text:
                    align_data["wrap_text"] = True
                if cell.alignment.shrink_to_fit:
                    align_data["shrink_to_fit"] = True
                if cell.alignment.indent:
                    align_data["indent"] = cell.alignment.indent
                if cell.alignment.text_rotation:
                    align_data["text_rotation"] = cell.alignment.text_rotation
                if align_data:
                    fmt["alignment"] = align_data
            if cell.border is not None:
                border_data = {}
                for side_name in ("left", "right", "top", "bottom"):
                    side = getattr(cell.border, side_name)
                    if side.style is not None:
                        side_data = {"style": side.style}
                        if side.color is not None:
                            side_data["color"] = side.color
                        border_data[side_name] = side_data
                if border_data:
                    fmt["border"] = border_data
            if cell.fill is not None:
                fill_data = {}
                if cell.fill.fill_type is not None:
                    fill_data["fill_type"] = cell.fill.fill_type
                if cell.fill.start_color is not None:
                    fill_data["start_color"] = cell.fill.start_color
                if cell.fill.end_color is not None:
                    fill_data["end_color"] = cell.fill.end_color
                if fill_data:
                    fmt["fill"] = fill_data

            # Handle number format
            nf = cell.number_format
            if isinstance(cell.value, datetime) and nf == "General":
                nf = "yyyy-mm-dd hh:mm:ss"
            elif isinstance(cell.value, date) and not isinstance(cell.value, datetime) and nf == "General":
                nf = "yyyy-mm-dd"
            if nf != "General":
                fmt["number_format"] = nf

            if fmt:
                rb.set_cell_format(self._sheet_idx, row - 1, col - 1, json.dumps(fmt))

        # Column widths
        for letter, dim in self.column_dimensions.items():
            if dim.width is not None:
                _, col_idx = _parse_cell_ref(f"{letter}1")
                rb.set_column_width(self._sheet_idx, col_idx - 1, dim.width)

        # Row heights
        for row_num, dim in self.row_dimensions.items():
            if dim.height is not None:
                rb.set_row_height(self._sheet_idx, row_num - 1, dim.height)

        # Freeze panes
        if self.freeze_panes:
            r, c = _parse_cell_ref(self.freeze_panes)
            rb.set_freeze_panes(self._sheet_idx, r - 1, c - 1)
```

```rust
fn build_format_from_json(json_str: &str) -> Result<Format, PyErr> {
    let data: serde_json::Value = serde_json::from_str(json_str)
        .map_err(|e| pyo3::exceptions::PyValueError::new_err(e.to_string()))?;

    let mut fmt = Format::new();

    if let Some(font) = data.get("font").and_then(|v| v.as_object()) {
        if font.get("bold").and_then(|v| v.as_bool()).unwrap_or(false) {
            fmt = fmt.set_bold();
        }
        if font.get("italic").and_then(|v| v.as_bool()).unwrap_or(false) {
            fmt = fmt.set_italic();
        }
        if let Some(ul) = font.get("underline").and_then(|v| v.as_str()) {
            if ul == "single" {
                fmt = fmt.set_underline(rust_xlsxwriter::FormatUnderline::Single);
            } else if ul == "double" {
                fmt = fmt.set_underline(rust_xlsxwriter::FormatUnderline::Double);
            }
        }
        if let Some(name) = font.get("name").and_then(|v| v.as_str()) {
            fmt = fmt.set_font_name(name);
        }
        if let Some(size) = font.get("size").and_then(|v| v.as_f64()) {
            fmt = fmt.set_font_size(size);
        }
        if let Some(color) = font.get("color").and_then(|v| v.as_str()) {
            if let Ok(rgb) = u32::from_str_radix(color, 16) {
                fmt = fmt.set_font_color(rust_xlsxwriter::Color::from(rgb));
            }
        }
    }

    if let Some(align) = data.get("alignment").and_then(|v| v.as_object()) {
        if let Some(h) = align.get("horizontal").and_then(|v| v.as_str()) {
            let a = match h {
                "center" => rust_xlsxwriter::FormatAlign::Center,
                "right" => rust_xlsxwriter::FormatAlign::Right,
                "left" => rust_xlsxwriter::FormatAlign::Left,
                "fill" => rust_xlsxwriter::FormatAlign::Fill,
                "justify" => rust_xlsxwriter::FormatAlign::Justify,
                "centerContinuous" | "center_continuous" => rust_xlsxwriter::FormatAlign::CenterAcross,
                "distributed" => rust_xlsxwriter::FormatAlign::Distributed,
                _ => rust_xlsxwriter::FormatAlign::General,
            };
            fmt = fmt.set_align(a);
        }
        if let Some(v) = align.get("vertical").and_then(|v| v.as_str()) {
            let a = match v {
                "center" => rust_xlsxwriter::FormatAlign::VerticalCenter,
                "top" => rust_xlsxwriter::FormatAlign::Top,
                "bottom" => rust_xlsxwriter::FormatAlign::Bottom,
                "justify" => rust_xlsxwriter::FormatAlign::VerticalJustify,
                "distributed" => rust_xlsxwriter::FormatAlign::VerticalDistributed,
                _ => rust_xlsxwriter::FormatAlign::Bottom,
            };
            fmt = fmt.set_align(a);
        }
        if align.get("wrap_text").and_then(|v| v.as_bool()).unwrap_or(false) {
            fmt = fmt.set_text_wrap();
        }
        if align.get("shrink_to_fit").and_then(|v| v.as_bool()).unwrap_or(false) {
            fmt = fmt.set_shrink();
        }
        if let Some(ind) = align.get("indent").and_then(|v| v.as_u64()) {
            if ind > 0 { fmt = fmt.set_indent(ind as u8); }
        }
        if let Some(rot) = align.get("text_rotation").and_then(|v| v.as_i64()) {
            if rot != 0 { fmt = fmt.set_rotation(rot as i16); }
        }
    }

    if let Some(border) = data.get("border").and_then(|v| v.as_object()) {
        fn parse_bs(s: &str) -> rust_xlsxwriter::FormatBorder {
            match s {
                "thin" => rust_xlsxwriter::FormatBorder::Thin,
                "medium" => rust_xlsxwriter::FormatBorder::Medium,
                "thick" => rust_xlsxwriter::FormatBorder::Thick,
                "dashed" => rust_xlsxwriter::FormatBorder::Dashed,
                "dotted" => rust_xlsxwriter::FormatBorder::Dotted,
                "double" => rust_xlsxwriter::FormatBorder::Double,
                "hair" => rust_xlsxwriter::FormatBorder::Hair,
                _ => rust_xlsxwriter::FormatBorder::Thin,
            }
        }
        fn parse_clr(c: &str) -> Option<rust_xlsxwriter::Color> {
            u32::from_str_radix(c, 16).ok().map(rust_xlsxwriter::Color::from)
        }

        if let Some(left) = border.get("left").and_then(|v| v.as_object()) {
            if let Some(s) = left.get("style").and_then(|v| v.as_str()) { fmt = fmt.set_border_left(parse_bs(s)); }
            if let Some(c) = left.get("color").and_then(|v| v.as_str()) { if let Some(clr) = parse_clr(c) { fmt = fmt.set_border_left_color(clr); } }
        }
        if let Some(right) = border.get("right").and_then(|v| v.as_object()) {
            if let Some(s) = right.get("style").and_then(|v| v.as_str()) { fmt = fmt.set_border_right(parse_bs(s)); }
            if let Some(c) = right.get("color").and_then(|v| v.as_str()) { if let Some(clr) = parse_clr(c) { fmt = fmt.set_border_right_color(clr); } }
        }
        if let Some(top) = border.get("top").and_then(|v| v.as_object()) {
            if let Some(s) = top.get("style").and_then(|v| v.as_str()) { fmt = fmt.set_border_top(parse_bs(s)); }
            if let Some(c) = top.get("color").and_then(|v| v.as_str()) { if let Some(clr) = parse_clr(c) { fmt = fmt.set_border_top_color(clr); } }
        }
        if let Some(bottom) = border.get("bottom").and_then(|v| v.as_object()) {
            if let Some(s) = bottom.get("style").and_then(|v| v.as_str()) { fmt = fmt.set_border_bottom(parse_bs(s)); }
            if let Some(c) = bottom.get("color").and_then(|v| v.as_str()) { if let Some(clr) = parse_clr(c) { fmt = fmt.set_border_bottom_color(clr); } }
        }
    }

    if let Some(fill) = data.get("fill").and_then(|v| v.as_object()) {
        if let Some(ft) = fill.get("fill_type").and_then(|v| v.as_str()) {
            let pattern = match ft {
                "solid" => rust_xlsxwriter::FormatPattern::Solid,
                "darkGray" => rust_xlsxwriter::FormatPattern::DarkGray,
                "mediumGray" => rust_xlsxwriter::FormatPattern::MediumGray,
                "lightGray" => rust_xlsxwriter::FormatPattern::LightGray,
                _ => rust_xlsxwriter::FormatPattern::Solid,
            };
            fmt = fmt.set_pattern(pattern);
        }
        if let Some(sc) = fill.get("start_color").and_then(|v| v.as_str()) {
            if let Ok(rgb) = u32::from_str_radix(sc, 16) {
                fmt = fmt.set_background_color(rust_xlsxwriter::Color::from(rgb));
            }
        }
        if let Some(ec) = fill.get("end_color").and_then(|v| v.as_str()) {
            if let Ok(rgb) = u32::from_str_radix(ec, 16) {
                fmt = fmt.set_foreground_color(rust_xlsxwriter::Color::from(rgb));
            }
        }
    }

    if let Some(nf) = data.get("number_format").and_then(|v| v.as_str()) {
        if nf != "General" {
            fmt = fmt.set_num_format(nf);
        }
    }

    Ok(fmt)
}
```

```python
def load_workbook(filename, data_only=True):
    """Load an xlsx file. Returns a Workbook with cell values (no formatting)."""
    from openpyxl_rust._openpyxl_rust import _load_workbook

    data = _load_workbook(str(filename))
    sheet_names = list(data["sheet_names"])
    sheets_data = data["sheets"]

    wb = Workbook()
    wb._sheets = []
    # Reset Rust workbook — remove default sheet
    from openpyxl_rust._openpyxl_rust import RustWorkbook
    wb._rust_wb = RustWorkbook()
    # Remove the default "Sheet" by creating a fresh RustWorkbook
    # We need to manually rebuild it with no default sheet
    # Workaround: the RustWorkbook starts with one sheet, we'll overwrite it

    first = True
    for name in sheet_names:
        if first:
            # Reuse the default sheet (index 0)
            wb._rust_wb.set_sheet_title(0, name)
            ws = Worksheet(title=name, workbook=wb, sheet_idx=0)
            first = False
        else:
            idx = wb._rust_wb.add_sheet(name)
            ws = Worksheet(title=name, workbook=wb, sheet_idx=idx)

        rows = sheets_data[name]
        for r_idx, row in enumerate(rows):
            for c_idx, value in enumerate(row):
                if value is not None:
                    ws.cell(row=r_idx + 1, column=c_idx + 1, value=value)
        wb._sheets.append(ws)

    return wb
```
